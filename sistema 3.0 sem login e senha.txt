        // Funções para controle do calendário
        function gerarCalendario(ano, mes) {
            const diasNoMes = new Date(ano, mes + 1, 0).getDate();
            const primeiroDia = new Date(ano, mes, 1).getDay();
            
            let html = '';
            // Dias vazios no início
            for (let i = 0; i < primeiroDia; i++) {
                html += '<div style="height: 100px; background: #f9f9f9;"></div>';
            }
            
            // Dias do mês
            for (let dia = 1; dia <= diasNoMes; dia++) {
                const hoje = new Date();
                const isToday = dia === hoje.getDate() && mes === hoje.getMonth() && ano === hoje.getFullYear();
                
                html += `
                    <div style="height: 100px; border: 1px solid #eee; padding: 5px; 
                                ${isToday ? 'background: #e6f2ff;' : ''}">
                        <div style="text-align: right;">${dia}</div>
                        <!-- Eventos seriam adicionados aqui -->
                    </div>
                `;
            }
            
            document.getElementById('calendario').innerHTML = 
                document.querySelector('#calendario div').outerHTML + html;
            document.getElementById('current-month').textContent = 
                new Date(ano, mes).toLocaleDateString('pt-BR', { month: 'long', year: 'numeric' });
        }
